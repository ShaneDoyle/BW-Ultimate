// Have more control over how AI Gods defend themselves.
begin script CustomHardMode_GodAttack(PlayerNum)
	LoopForever = 1
	Attack = 0

	PlayerCitadel = get CITADEL at player PlayerNum temple position radius 100
	PlayerInfluence = 0
	
	MyInfluence = 0 //Get influence of PlayerNum
	OtherInfluence = 0 //Get influence of other player(s).
	isAlly = 0 // If both gods are allies, don't attack.

	TargetCreature = 0
	TargetPlayer = 1
	
	Attack_Spell = 0
	Lightning_Spell = 0
	MegaBlast_Spell = 0
	Freeze_Spell = 0 // This is a hardcoded event. It looks like AI has a hierarchy of spells they like to use, with freeze being near the bottom. 
	Freeze_Cooldown = 0
	RequiredMana = 0
	
	AttackSpellToUse = 0
	CreatureSpellToUse = 0
	

start
	wait 30 seconds // Wait 30 seconds before starting.
	
	// Give 1 time mana boost at start (if they have attack spells).
	Lightning_Spell=get spell icon SPELL_LIGHTNING_LEVEL_1 in PlayerCitadel
	if Lightning_Spell exists
		RequiredMana = get mana for spell SPELL_LIGHTNING_LEVEL_1*25
		set Lightning_Spell mana RequiredMana
	end if
	
	MegaBlast_Spell=get spell icon SPELL_EXPLOSION_LEVEL_1 in PlayerCitadel
	if MegaBlast_Spell exists
		RequiredMana = get mana for spell SPELL_EXPLOSION_LEVEL_1*15
		set MegaBlast_Spell mana RequiredMana
	end if
	
	while LoopForever == 1
		Attack = 0
		set computer player PlayerNum personality "ReactToAggressiveCreature" 1 // Don't let AI attack, we handle that here.
		
		if dialogue ready
			if TargetPlayer != PlayerNum
				if LandNumber == 2 and PlayerNum == 3 and TargetPlayer == 2 // Exception case. Do nothing if Lethys is attacking Khazar's Creature on Land 2.
					// Do nothing
				else
					isAlly = get player TargetPlayer ally percentage with player PlayerNum
					TargetCreature = get player TargetPlayer creature
					wait 0.3 seconds
					
					if TargetCreature exists and isAlly == 0
						PlayerInfluence = get player PlayerNum raw influence at [TargetCreature]
						if PlayerInfluence > 0
							if Attack == 0 and HEALTH of TargetCreature != 0
							
								// Sort of cheat this... oh well...
								clear computer player PlayerNum actions
								set computer player PlayerNum position to [TargetCreature] with fixed height
								wait 1 seconds
								Freeze_Spell = get spell icon CREATURE_SPELL_FREEZE in PlayerCitadel
								if Freeze_Spell exists or LandNumber == 5 // Just cheat this for Nemesis.
									if HEALTH of TargetCreature != 0
										Freeze_Spell = cast CREATURE_SPELL_FREEZE spell on TargetCreature from computer player PlayerNum position radius 0 time 60 curl 0
									end if
								end if
								
								release computer player PlayerNum
								clear computer player PlayerNum actions
								force computer player PlayerNum action "AttackCreatureWithSpells" TargetCreature
								Attack = 1
								wait 20 seconds
							end if
						end if
					end if
				end if	
			end if
		end if
		
		// Check next Player.
		TargetPlayer++
		if TargetPlayer > 7
			TargetPlayer = 1
		end if
		
		wait 0.1 seconds
	end while


end script CustomHardMode_GodAttack

// Make AI creature capture towns.
begin script CustomHardMode_CreatureCaptureTown(PlayerNum, TownNum)
	Town = get town with id TownNum
	TownStore = get STORE in Town
	Creature = 0 //get player 3 creature
	TotalBelief = 0
	ImpressTimer = 1800 // 30 mins. Put in timer just INCASE something fucks up...
	Retries = 500
start
	Creature = get player PlayerNum creature
	wait 0.3 seconds
	
	
	while Retries > 0
		if get PLAYER of Town <= 0
			if size of Town > 0 // Check if town not dead. Should also cover if town doesn't exist.
				
				//Queue up Creature to impress.
				queue computer player PlayerNum action "GetCreaturetoHelp" Town Creature
				wait until computer player PlayerNum ready
				release computer player PlayerNum
				wait 5 seconds
				Retries = 0
				
				// Wait until town is not neutral or town dies and if the leash object exists (village store). Also, periodically add some extra belief.
				while get PLAYER of Town <= 0 and size of Town > 0 and Creature leashed and ImpressTimer > 0
					
					//Remove leash if Creature dead.
					if HEALTH of Creature == 0
						detach Creature leash
						release Creature
					end if
					
					wait 1 seconds
					ImpressTimer -= 1
				end while
				
				//Release the Creature.
				detach Creature leash
				release Creature
			end if
		end if
		wait 0.1 seconds
		Town = get town with id number from 0 to 13
		Retries -= 1
	end while
	
end script CustomHardMode_CreatureCaptureTown

begin script CustomHardMode_CreatureAttackPlayerTemple(PlayerNum)
	Creature = 0
	Citadel = PlayerNum
	PlayerToAttack = PlayerNum
	LeashType = 0
	AttackTimer = 600
	
start
	Creature = get player PlayerNum creature
	wait 1.0 seconds
	
	//Select a player to attack. (That isn't ourself). TO-DO: Need to add Ally support here
	while PlayerToAttack == PlayerNum
		PlayerToAttack = number from 1 to 4
	end while
	
	//Get the temple.
	Citadel = get CITADEL at player PlayerToAttack temple position radius 100
	wait 1.0 seconds
	
	//Queue up Creature to attack the citadel.
	//clear computer player PlayerNum actions
	queue computer player PlayerNum action "AttackCreatureWithCreature" MyCreature Creature
	wait until computer player PlayerNum ready
	release computer player PlayerNum
	wait 2 seconds
	
	//Attach creature to the temple.
	attach Creature leash to Citadel
	
	
	//While angry leash and attached.
	while Creature leashed and AttackTimer > 0
		AttackTimer -= 1
		wait 1 second
	end while
	
	wait 180 seconds
	
	detach Creature leash
	release Creature
	

end script CustomHardMode_CreatureAttackPlayerTemple

begin script CustomHardMode_ImproveAI_AttackTowns(PlayerNum)
	Max_Towns = 0
	Target_Town = 0
	i = 0
	Town_Owner = 0
	Lightning_Spell = 0
	MegaBlast_Spell = 0
	PlayerCitadel = get CITADEL at player PlayerNum temple position radius 100
	PlayerInfluence = 0
	RequiredMana = 0
	TownStore = 0
	Attack = 0

start
	
	// Don't start this attacking behaviour for 5 mins.
	//wait 60 * 5 seconds
	wait 45 seconds
	
	while Difficulty == 1
		while i <= 20 // Random number, 20 is probably good for Village caps.
			Target_Town = get town with id i
			Town_Owner = get PLAYER of Target_Town
			Attack = 0
			
			// If town is not our own, not neutral and has population, do some attacking.
			if Town_Owner != PlayerNum
				if Town_Owner != 0
					if size of Target_Town > 0
						TownStore = get STORE in Target_Town
						if TownStore exists
							PlayerInfluence = get player PlayerNum raw influence at [TownStore]
							if PlayerInfluence > 0
							
								MegaBlast_Spell=get spell icon SPELL_EXPLOSION_LEVEL_1 in PlayerCitadel
								if MegaBlast_Spell exists and Attack == 0
									queue computer player PlayerNum action "DestroytownWithSpell" Target_Town MegaBlast_Spell
									Attack = 1
									wait 60 seconds
								end if
								
								Lightning_Spell=get spell icon SPELL_LIGHTNING_LEVEL_1 in PlayerCitadel
								if Lightning_Spell exists and Attack == 0
									queue computer player PlayerNum action "DestroytownWithSpell" Target_Town Lightning_Spell
									Attack = 1
									wait 60 seconds
								end if
							end if
						end if
					end if
				end if
			end if
			i++
			wait 1 second
		end while
		i = 0
	end while

end script CustomHardMode_ImproveAI_AttackTowns



//Make Gods Impress Towns + Attack Players.
begin script CustomHardMode_ImproveAIController(PlayerNum)
	Max_Towns = 0
	Target_Town = 11
	ComputerAIAction = 0
	Creature = 0
	Choose = 0

start
	Creature = get player PlayerNum creature
	wait 5 seconds
	
	wait until dialogue ready // Wait until intro to land has finished.
		
	while Difficulty == 1
		
		//If no action selected. Main controller of creature action selector.
		if ComputerAIAction == 0
		
			//Randomise selection.
			Choose = number from 1 to 100
			
			//85% chance to impress village.
			if Choose < 85
				//Max_Towns = get neutral town total
				Target_Town = number from 0 to 13
				ComputerAIAction = 1
				
			//10% chance to attack town.
			elsif Choose < 95
				Target_Town = number from 0 to 13
				ComputerAIAction = 2
				
			//5% chance to attack citadel.
			else
				ComputerAIAction = 3
			end if
			
		end if
		
		// Impress neutral towns.
		if ComputerAIAction == 1
			run script CustomHardMode_CreatureCaptureTown(PlayerNum, Target_Town)
			ComputerAIAction = 0
		end if
		
		//Attack another player's town.
		if ComputerAIAction == 2
			run script CustomHardMode_CreatureAttackPlayerTown(PlayerNum, Target_Town)
			ComputerAIAction = 0
		end if
		
		//Attack another player's temple.
		if ComputerAIAction == 3
			run script CustomHardMode_CreatureAttackPlayerTemple(PlayerNum)
			ComputerAIAction = 0
		end if
		
		//Remove the actions. Should fix the bug with creature constantly leashing over and over...
		// clear computer player PlayerNum actions
		
		//Small delay between decisions.
		wait 30 second
		
	end while

end script CustomHardMode_ImproveAIController
