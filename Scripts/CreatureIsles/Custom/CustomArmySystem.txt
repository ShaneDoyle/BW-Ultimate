begin script OgreAlignment(Creature)
start
	while 1 == 1
		ALIGNMENT of Creature = HEALTH of Creature * -1
	end while
	
end script OgreAlignment

begin script KillOgre(Creature)
start
	while 1 == 1
		wait until HEALTH of Creature <= 0
		force Creature CREATURE_DIE_PERMANENTLY Creature
		wait until Creature played
	end while
	
end script KillOgre

begin script KeepRockNextToCreature
	Dist = -5
	Ref = 0
	Rock = create OBJECT ROCK at [1071.06, 132.618, 1207.3]
	CX = 0
	CY = 0
	CZ = 0
	RX = 0
	RZ = 0
	DX = 0
	DZ = 0
	L = 0
	Creature = get player 1 creature
	TykeHeight = SCRIPT_OBJECT_PROPERTY_TYPE_HEIGHT of Creature
start
	SCALE of Rock = 0
	begin loop
		Dist = -20
		Ref = marker at belly position of MyCreature
		Creature = get player 1 creature
		TykeHeight = SCRIPT_OBJECT_PROPERTY_TYPE_HEIGHT of Creature
		CX = XPOS of Creature
		CY = ALTITUDE of Creature
		CZ = ZPOS of Creature
		RX = XPOS of Ref
		RZ = ZPOS of Ref
		DX = RX - CX
		DZ = RZ - CZ
		L = sqrt(DX * DX + DZ * DZ)
		DX = DX / L * Dist
		DZ = DZ / L * Dist
		set Rock position to [CX + DX, TykeHeight + 10, CZ + DZ]
		ALTITUDE of Rock = TykeHeight + 4
		//move camera position [CX + DX, 0, CZ + DZ] time 3
		begin cinema
			move camera position to [Rock] time 8
			move camera focus to [Creature] + [0, 5, 0] time 8
		end cinema
	end loop
end script KeepRockNextToCreature

begin script CreatureStomp(Creature, Unit, Action)
	ThrowPos = marker at [0,0,0]
	XThrow = 0
	XBuffer = 0
	ZThrow = 0
	ZBuffer = 0
	Buff = 0
	CreatureRotation = 0
start

	//start sound constant from LH_SAMPLE_G_SQUASHANIMAL_01 to LH_SAMPLE_G_SQUASHANIMAL_04 AUDIO_SFX_BANK_TYPE_SCRIPT_SFX at [Creature]
	
	//start sound LH_SAMPLE_SKITTLECOLLIDE_01
	if ArmyCreatures[0] != Creature and HEALTH of Creature > 0.1
		ArmyCreatures[0] = Creature
		
		// Stomp
		if Action == 1
			force Creature CREATURE_STOMP Unit
			HEALTH of Unit = 0
		elsif Action == 2
			force Creature CREATURE_STOMP Unit
			HEALTH of Unit = 0
		//Throw!
		/*
		elsif Action == 0
			wait until Creature played 
			force Creature CREATURE_IDLE Creature
			wait 1.5 second
			release Creature
			Creature play C_INDIVIDUAL_SUMMON
			wait 1.5 second
			Buff = cast MAGIC_TYPE_EXPLOSION_ONE spell on Creature from [Creature] radius 0 time 15 curl 0
		*/
		//Throw!
		elsif Action == 3 or Action == 4
			//force Creature CREATURE_KICK_TREE Unit
			//queue Creature fight move CTR_HIT_BLOCK
			ThrowPos = marker at [Unit] + [50,-100,50]
			//ThrowPos = marker at [(get XPOS of Creature)-(10(cos (EpicBuildingRotation))/57.296), -50 ,(get ZPOS of Creature)-(10(sin (EpicBuildingRotation))/57.296)]
			CreatureRotation = get ANGLE of Creature
			
			XThrow = number from -50 to 50
			if XThrow < 20
				XThrow = 20
			end if
			if XThrow > -20
				XThrow = -20
			end if
			
			if ZThrow < 25
				XThrow = 25
			end if
			if ZThrow > -25
				XThrow = -25
			end if
			
			ZThrow = number from -50 to 50
			
			ThrowPos = marker at [XPOS of Unit + XThrow, -50, ZPOS of Unit +ZThrow]
			force Creature CREATURE_HOLD_OBJECT Unit
			wait until Unit in Creature hand or 5 seconds
			force Creature CREATURE_HURL_OBJECT_IN_HAND ThrowPos with Unit
			wait until Creature played or Unit not in Creature hand or 5 seconds
			
			/*
			
					force MyCreature CREATURE_HOLD_OBJECT Marble
					wait until Marble in MyCreature hand
					force MyCreature CREATURE_HURL_OBJECT_IN_HAND AimingPoint with Marble
					*/
		// Kick
		elsif Action == 5
			//force Creature CREATURE_CAST_FIREBALL Unit
			force Creature CREATURE_KICK Unit
		
		end if
		wait until Creature played or 5 seconds
		release Creature
		ArmyCreatures[0] = -1
	end if
end script CreatureStomp


begin script NomanFollow(Unit, Target, x_adjust, z_adjust, i)
	NewScore = 0
	j = i
	TargetIsFriendly = 0 //0 Enemy, 1 Friend.
	Mist = 0
	DeathTimer = 60
	IntoPositionDelay = 10
start
	z_adjust = -2
	x_adjust = (i % 10) - 5
	while j >= 10
		j-=10
		z_adjust++
	end while
	
	set Unit position to [Target] + [x_adjust, z_adjust]
	SCALE of Unit = 1.5
	HEALTH of Unit = 0.25
	
	//set Unit anim ANM_P_ON_FIRE_RUN
	while HEALTH of Unit > 0
		SPEED of Unit = 0.75
		delete Mist
		Mist = create mist at [Unit] scale 0.1 red 240 green 0 blue 0 transparency 0.2 height ratio 1 
		if ArmyAction == 1
			if get distance from [AttackTarget] to [Unit] >= 6
				move Unit position to [(get XPOS of AttackTarget)-(4*(cos (i*18))/57.296), 0 ,(get ZPOS of AttackTarget)-(4*(sin (i*18))/57.296)]
			elsif IntoPositionDelay > 0
				move Unit position to [(get XPOS of AttackTarget)-(4*(cos (i*18))/57.296), 0 ,(get ZPOS of AttackTarget)-(4*(sin (i*18))/57.296)]
				IntoPositionDelay-= 1
			else
				if TargetIsFriendly == 1
					SPEED of Unit = 3
				else
					start music MUSIC_TYPE_SCRIPT_NEMESIS
					disable alignment music
					HEALTH of AttackTarget -=0.0004 
					run background script CreatureStomp(AttackTarget, Unit, number from 4 to 5)
					
					set Unit focus to [AttackTarget]
					wait number from 0.1 to 1.5 seconds
					Unit play ANM_P_SLEDGEHAMMER
					wait number from 1.2 to 2 seconds
					
				end if
			end if
		else
			move Unit position to [Target] + [x_adjust, z_adjust]
		end if
		
	end while
	
	
	delete Mist
	
	
	NewScore = get Target score 
	set Target score NewScore - 1
	
	wait 15 seconds
	
	while DeathTimer > 0
		SCALE of Unit-= 0.025
		DeathTimer--
	end while
	delete Unit
end script NomanFollow
