begin script DoBeams(Origin, Target)

	Distance = get distance from [Origin] to [Target]

		ExplosionTarget = marker at (get target from [Origin] - ([Target] - [Origin]) to [Origin] distance Distance * Step angle 0)
		Explosion = create special effect SPOT_VISUAL_SEE_THIS_BEAM at [ExplosionTarget] time 10

	end while

begin script CreateWinIndicator(Creature)
	Count = 0
	CreatureSubType = 0
	Mushroom = 0
	CreatureFound = 0
	Plinth = 0
	RandomAngle = number from 0 to 360
	AlreadyPlacedObj = 0
start
	CreatureSubType = variable get Creature sub type
	while Count < 18 and CreatureFound == 0
		if CreatureSubType == Creatures[Count]
			Plinth = CreaturePortraits[Count]
			CreatureFound = 1
		end if
		Count++
	end while
	run script GetAngleForPlinth(Plinth, 59)
	AlreadyPlacedObj = get FEATURE FEATURE_INFO_MAGIC_MUSHROOM at [CloseUpOfPlinth]
	if not AlreadyPlacedObj exists
		Mushroom = create with angle RandomAngle and scale 0.5 FEATURE FEATURE_INFO_MAGIC_MUSHROOM at [CloseUpOfPlinth]
		disable Mushroom pickup
		enable Mushroom indestructible
		disable Mushroom hurt by fire
	end if
end script CreateWinIndicator
