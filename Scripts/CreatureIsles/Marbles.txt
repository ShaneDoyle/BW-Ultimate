begin script CalculateTargetPos
	Angle = 0
	Distance = 0
	DistanceToCentre = 0
	MarblePos = 0
	MarbleDistance = 0
	MarbleDistanceOut = 0
	MarbleDistanceFromLeftEdge = 0
	MarbleDistanceFromRightEdge = 0
	MarbleAngleOut = 0
	LeftEdgePos = 0
	RightEdgePos = 0
	Sparkle = 0
start
	if MarblesCancelCurrentGame == 0
		LeftEdgePos = marker at (get target from [MarblePoint1] to [CentreOfArena] distance 42.5 angle -90)
		RightEdgePos = marker at (get target from [MarblePoint1] to [CentreOfArena] distance 42.5 angle 90)
		if Round == 1
			if not CentreOfArenaRock exists
				CentreOfArenaRock = create with angle 0 and scale 0.001 ROCK_OBJECT ROCK at [CentreOfArena]
				enable CentreOfArenaRock indestructible
				disable CentreOfArenaRock pickup
				disable CentreOfArenaRock moveable
			end if
			if not ArenaLeftEdge exists
				ArenaLeftEdge = create with angle 0 and scale 0.001 ROCK_OBJECT ROCK at [LeftEdgePos]
				enable ArenaLeftEdge indestructible
				disable ArenaLeftEdge pickup
				disable ArenaLeftEdge moveable
			end if
			if not ArenaRightEdge exists
				ArenaRightEdge = create with angle 0 and scale 0.001 ROCK_OBJECT ROCK at [RightEdgePos]
				enable ArenaRightEdge indestructible
				disable ArenaRightEdge pickup
				disable ArenaRightEdge moveable
			end if
		end if
		DistanceToCentre = get distance from [CentreOfArena] to [ApeThrowPos]
		if Round > 1
			if Round == 2
				MarblePos = marker at [OMarble[0]]
			elsif Round == 3
				MarblePos = marker at [OMarble[1]]
			elsif Round == 4
				MarblePos = marker at [OMarble[2]]
			elsif Round == 5
				MarblePos = marker at [OMarble[3]]
			elsif Round == 6
				MarblePos = marker at [OMarble[4]]
			end if
			MarbleDistance = get distance from [MarblePos] to [ApeThrowPos]
			MarbleDistanceOut = get distance from [MarblePos] to [CentreOfArena]
			if DistanceToCentre > MarbleDistance
				EstimatedAmountOfRoll += MarbleDistanceOut / 2
			elsif DistanceToCentre < MarbleDistance
				EstimatedAmountOfRoll -= MarbleDistanceOut / 2
			else
			end if
			MarbleDistanceFromLeftEdge = get distance from [MarblePos] to [ArenaLeftEdge]
			MarbleDistanceFromRightEdge = get distance from [MarblePos] to [ArenaRightEdge]
			MarbleAngleOut = get distance from [MarblePos] to [CentreOfArena]
			if MarbleDistanceFromRightEdge > MarbleDistanceFromLeftEdge
				EstimatedAngle += MarbleAngleOut / 2
			elsif MarbleDistanceFromRightEdge < MarbleDistanceFromLeftEdge
				EstimatedAngle -= MarbleAngleOut / 2
			else
			end if
		end if
		if EstimatedAmountOfRoll > MarblesMaxRoll
			EstimatedAmountOfRoll = MarblesMaxRoll
		end if
		if EstimatedAmountOfRoll < MarblesMinRoll
			EstimatedAmountOfRoll = MarblesMinRoll
		end if
		if EstimatedAngle > MarblesMaxAngle
			EstimatedAngle = MarblesMaxAngle
		end if
		if EstimatedAngle < MarblesMinAngle
			EstimatedAngle = MarblesMinAngle
		end if
		Distance = DistanceToCentre - EstimatedAmountOfRoll
		Angle = EstimatedAngle
		ThrowTargetPos = marker at (get target from [ApeThrowPos] to [CentreOfArena] distance Distance angle Angle)
	end if
end script CalculateTargetPos

begin script CalculateScore
	CurrentScore = 0
	Distance = 0
	OuterRingScore = 10
	OuterMiddleRingScore = 30
	InnerMiddleRingScore = 70
	InnerRingScore = 150
	Counter = 0
start
	while Counter < 6 and MarblesCancelCurrentGame == 0
		if PMarble[Counter] exists
			wait until not PMarble[Counter] is FLYING
		end if
		Counter++
		until MarblesCancelCurrentGame == 1
	end while
	Counter = 0
	while Counter < 6 and MarblesCancelCurrentGame == 0
		if OMarble[Counter] exists
			wait until not OMarble[Counter] is FLYING
		end if
		Counter++
		until MarblesCancelCurrentGame == 1
	end while
	Counter = 0
	while Counter < 6 and MarblesCancelCurrentGame == 0
		if PMarble[Counter] exists
			Distance = get distance from [CentreOfArena] to [PMarble[Counter]]
			if Distance <= InnerRing
				CurrentScore += InnerRingScore
			elsif Distance <= InnerMiddleRing
				CurrentScore += InnerMiddleRingScore
			elsif Distance <= OuterMiddleRing
				CurrentScore += OuterMiddleRingScore
			elsif Distance <= OuterRing
				CurrentScore += OuterRingScore
			else
				CurrentScore += 0
			end if
		end if
		Counter++
	end while
	MyScore = CurrentScore
	CurrentScore = 0
	Counter = 0
	while Counter < 6 and MarblesCancelCurrentGame == 0
		if OMarble[Counter] exists
			Distance = get distance from [CentreOfArena] to [OMarble[Counter]]
			if Distance <= InnerRing
				CurrentScore += InnerRingScore
			elsif Distance <= InnerMiddleRing
				CurrentScore += InnerMiddleRingScore
			elsif Distance <= OuterMiddleRing
				CurrentScore += OuterMiddleRingScore
			elsif Distance <= OuterRing
				CurrentScore += OuterRingScore
			else
				CurrentScore += 0
			end if
		end if
		Counter++
	end while
	ApeScore = CurrentScore
end script CalculateScore

	if MarbleTrialRunning == 1 and not CancelGameHighlight exists
		if MarblesTrialCompleted == 0
			CancelGameHighlight = create highlight CHALLENGE_NONE HIGHLIGHT_CANCEL_QUEST at [CancelGameHighlightPos]

		else

		end if
	end if
	clear clicked object
	while CancelConfirmedOrCancelled == 0 and MarblesCancelCurrentGame == 0 and MarbleTrialRunning == 1
		if CancelGameHighlight exists
			if CancelGameHighlight clicked
				if ClickedOnce == 0
					ClickedOnce = 1
					wait until 0.1 seconds
					clear clicked object
					set ConfirmTimer time to 30 seconds
					run background script MarblesConfirmCancelMessage
				elsif ClickedOnce == 1
					MarblesCancelCurrentGame = 1

					ClickedOnce = 0
				end if
			end if
		else

		end if
	end while

		delete CancelGameHighlight
	end if
