					if RandomHit == 1

				elsif RandomForcePower == 2

				elsif RandomForcePower == 3

				elsif RandomForcePower == 4
					run script ForceThrow
				end if
			elsif DoSpellCast == True

				Spell = True
			end if

				queue ZediCreature fight move constant NextFightMoveForZedi strength 0.5
			elsif Defend == True
				queue ZediCreature fight move constant NextFightMoveForZedi strength 0
			elsif Special == True
				queue ZediCreature fight move constant NextFightMoveForZedi strength 1
			elsif Spell == True
				queue ZediCreature fight spell constant NextSpellMoveForZedi
			end if
			if ZediWinning == True
				if MyCreatureHealth < 0.8 and MyCreatureHealth > 0.2

							wait until not say sound constant WinLose playing

					end if
				end if
			else
				if ZediHealth < 0.8 and ZediHealth > 0.2
					if get WinLoseTimer time remaining <= 0

							wait until not say sound constant WinLose playing

					end if
				end if
			end if
		end if
	end while

	clear MyCreature fight queue
	enable ZediCreature auto fighting
	enable MyCreature auto fighting

	NegativeAlignment = 0

				NegativeAlignment = 1
			else
				ALIGNMENT of ZediCreature = StartingAlignment
			end if
		else
			MorphDone = 1
		end if
	end while

	end cinema with widescreen
end script PlayerWonFight

begin script PlayerLostFight
	ZediHeight = 0
	ZediCreaturesFightPos = 0
	Marker = 0
	EndOfFightMarker = 0
	EndOfFightCameraPos = 0
	CloseUpOfZedi = 0
	ZediPosAtEndOfFight = 0
	CamPosFollowingZedi = 0
	HardLuck = 0
	MoveRhinoToPos = 0
	LookOverZedisShoulder = 0
	RhinoStillWalking = 0
	DistanceBetweenCreatures = 0
	OldDistanceBetweenCreatures = 0
	MidPoint = 0
	Offset = 0
	CameraEndPos = 0
	BetweenScrollAndFightPos = marker at [2760.9775, 22.11, 1204.5294]
	InitialCameraPos = 0
	CamFinishPos = marker at [2872.8564, 81.1397, 1077.0604]
	CamFinishFoc = marker at [2715.0312, 21.446, 1178.5648]
start
	Marker = marker at [ZediFightArenaPos] + ([10, 10])
	ZediCreaturesFightPos = marker at (get target from [Marker] to [ZediFightArenaPos] distance 70 angle 180)
	EndOfFightMarker = marker at [ZediCreature]
	EndOfFightCameraPos = marker at (camera position)
	CloseUpOfZedi = marker at (get target from [EndOfFightCameraPos] to [EndOfFightMarker] distance 30 angle 180)
	ZediHeight = HEIGHT of ZediCreature
	ZediHeight = ZediHeight * 0.9
	begin cinema
		force MyCreature CREATURE_DEAD_FOREVER MyCreature
		InitialCameraPos = marker at (get target from [MyCreature] to [ZediCreature] distance 50 angle 60)
		set camera position to [InitialCameraPos] + [0, ZediHeight * 2, 0]
		set camera focus to [MyCreature] + [0, ZediHeight, 0]
		begin dual camera to ZediCreature MyCreature
			ENERGY of ZediCreature = 1
			force ZediCreature CREATURE_CAST_HEAL_SPELL_PU1 MyCreature
			while RhinoStillWalking == 0
				wait until 3 seconds
				OldDistanceBetweenCreatures = get distance from [MyCreature] to [ZediCreature]
				if OldDistanceBetweenCreatures <= DistanceBetweenCreatures
					RhinoStillWalking = 1
				end if
				DistanceBetweenCreatures = OldDistanceBetweenCreatures
			end while
		end dual camera
		LookOverZedisShoulder = marker at (get target from [MyCreature] to [ZediCreature] distance ZediHeight angle 40)
		CameraEndPos = marker at (get target from [MyCreature] to [ZediCreature] distance ZediHeight angle 60)
		Offset = get distance from [MyCreature] to [ZediCreature]
		Offset /= 3
		MidPoint = marker at (get target from [MyCreature] to [ZediCreature] distance Offset angle 180)
		set camera position to [LookOverZedisShoulder] + [0, ZediHeight + ZediHeight / 2, 0]
		set camera focus to [MidPoint] + [0, ZediHeight / 2, 0]
		move camera position to [CameraEndPos] + [0, ZediHeight + ZediHeight / 2, 0] time 12
		disable creature sound
		HardLuck = variable constant from HELP_TEXT_ZEDI_FIGHT_RHINO_WON_01 to HELP_TEXT_ZEDI_FIGHT_RHINO_WON_03
		say single line constant HardLuck by ZediCreature
		wait until read
		close dialogue
		enable creature sound
		wait until ZediCreature played
		HEALTH of MyCreature = 1
		set ZediCreature focus to [MyCreature]
		ZediCreature play ANM_SWALLOW_FLAP loop 1
		wait until ZediCreature played
		move ZediCreature position to [ZediCreaturesFightPos] radius 0
		Marker = marker at [MyCreature]
		ZediPosAtEndOfFight = marker at [ZediCreature]
		CamPosFollowingZedi = marker at (get target from [ZediPosAtEndOfFight] to [Marker] distance 40 angle 0)
		move camera position to [CamFinishPos] time 5
		move camera focus to [CamFinishFoc] time 5
	end cinema
	wait until get distance from [ZediCreature] to [ZediCreaturesFightPos] < 1 or 10 seconds
	set ZediCreature focus to [ZediFightArenaPos]
end script PlayerLostFight

begin script LevitateRocksBetweenFights
	StartLevitatingTimer = create timer for 0 seconds
start
	while CreaturesFighting == 0
		if get StartLevitatingTimer time remaining <= 0
			wait until 10 seconds
			set StartLevitatingTimer time to 60 seconds
		end if
	end while
end script LevitateRocksBetweenFights

begin script RepositionThrownRock(RockNo)
	Scale = 0
	Angle = 0
	CreateRock = 0
	SFX = 0
	Smoke = 0
start
	wait until not ZediRocks[RockNo] exists or not ZediRocks[RockNo] is FLYING or not get distance from [ZediRocks[RockNo]] to [ZediFightArenaPos] < 150
	if ZediRocks[RockNo] exists
		if ZediRocks[RockNo] is FLYING
			CreateRock = 1
		else
			CreateRock = 0
		end if
	else
		CreateRock = 1
	end if
	if CreateRock == 1
		if ZediRocks[RockNo] exists
			delete ZediRocks[RockNo]
		end if
		Scale = number from 1.1 to 2.4
		Angle = number from 0 to 360
		if RockNo < 6
			ZediRocks[RockNo] = create with angle Angle and scale Scale ROCK_OBJECT ROCK at [FirstRockPilePos] + [number from -10 to 10, 0, number from -10 to 10]
		elsif RockNo < 12
			ZediRocks[RockNo] = create with angle Angle and scale Scale ROCK_OBJECT ROCK at [SecondRockPilePos] + [number from -10 to 10, 0, number from -10 to 10]
		else
			ZediRocks[RockNo] = create with angle Angle and scale Scale ROCK_OBJECT ROCK at [ThirdRockPilePos] + [number from -10 to 10, 0, number from -10 to 10]
		end if
	else
		if RockNo < 6
			set ZediRocks[RockNo] position to [FirstRockPilePos] + [number from -10 to 10, 0, number from -10 to 10]
		elsif RockNo < 12
			set ZediRocks[RockNo] position to [SecondRockPilePos] + [number from -10 to 10, 0, number from -10 to 10]
		else
			set ZediRocks[RockNo] position to [ThirdRockPilePos] + [number from -10 to 10, 0, number from -10 to 10]
		end if
	end if
	enable ZediRocks[RockNo] indestructible
	disable ZediRocks[RockNo] pickup
	SFX = create special effect SPOT_VISUAL_OBJECT_APPEAR at [ZediRocks[RockNo]] time 2
	Smoke = create special effect SPOT_VISUAL_BANG at [ZediRocks[RockNo]] time 2
end script RepositionThrownRock

begin script CheckForSpells
	ObjectInHand = 0
	TempType = 0
start
	begin loop
		ObjectInHand = get object held
		TempType = variable get ObjectInHand type
		if TempType == variable ONE_SHOT or TempType == variable SPELL_SEED
			delete ObjectInHand
		end if
		wait until 0.1 seconds
		until CreaturesFighting == 0
	end loop
end script CheckForSpells
