begin script CropStatus(Crop, Field)
	CropHeight = 0
	WateredToday = 0
	ManureAdded = 0
	ManureAddedFlag = 0
	Manure = 0
	LastTimeOfDay = 0
	Fireworks = 0
	FindWaterSpell = 0
	SpellInHand = 0
	MyHalfWayFlag = 0
	FieldScore = 0
	RandomSay = 0
	DayNum = 0
start
	while CropHeight < 4
		wait until 1 seconds
		if ManureAdded == 1
			ManureAddedFlag = 1
			ManureAdded = 2
		end if
		if get game time < LastTimeOfDay and SpiritActive == 0
			DayNum++
			wait until 2 seconds
			if WateredToday == 0
				if CropHeight > 0
					CropHeight--
					HEIGHT of Crop -= 0.2
					if Field == 1
						Crop1Total--
					else
						Crop2Total--
					end if
				end if
			elsif WateredToday < 3
				CropHeight++
				HEIGHT of Crop += 0.2
				if Field == 1
					Crop1Total++
				else
					Crop2Total++
				end if
			else
			end if
			WateredToday = 0
			if ManureAddedFlag == 1
				CropHeight = CropHeight + 1
				HEIGHT of Crop = (HEIGHT of Crop) + 0.2
				ManureAddedFlag = 0
				if Field == 1
					Crop1Total++
				else
					Crop2Total++
				end if
			end if
			if MyHalfWayFlag == 0 and CropHeight > 1
				FieldHalfWay++
				MyHalfWayFlag = 1
			elsif MyHalfWayFlag == 1 and CropHeight <= 2
				FieldHalfWay--
				MyHalfWayFlag = 0
			end if
			FieldScore = CropHeight * 2 + 2
			if Field == 1
				set CropScoreboard1 score FieldScore
			else
				set CropScoreboard2 score FieldScore
			end if
		end if
		LastTimeOfDay = get game time
		FindWaterSpell = get spell SPELL_WATER_LEVEL_1 at [Crop] radius 15
		if FindWaterSpell exists
			SpellInHand = get object held
			if WateredToday < 2
				begin dialogue
					eject good spirit
					RandomSay = variable constant from HELP_TEXT_SPECIAL_CROPS_291 to HELP_TEXT_SPECIAL_CROPS_294
					say single line constant RandomSay
					wait until read
					send good spirit home
				end dialogue
			else
				begin dialogue
					eject good spirit
					say HELP_TEXT_SPECIAL_CROPS_300
					wait until read
					say HELP_TEXT_SPECIAL_CROPS_310
					wait until read
					send good spirit home
				end dialogue
				OverWatered = 1
			end if
			while SpellInHand exists
			end while
			WateredToday++
			if LippyWaterCheck == 0
				LippyWaterCheck = 1
			end if
			wait until 2 seconds
		end if
		if ManureAdded == 0 or ManureAdded == 2
			Manure = get POO at [Crop] radius 10
			if Manure exists
				if ManureAdded == 0
					begin dialogue
						eject evil spirit
						say HELP_TEXT_SPECIAL_CROPS_320
						wait until read
						say HELP_TEXT_SPECIAL_CROPS_330
						wait until read
						send evil spirit home
					end dialogue
					delete Manure with fade
					wait until not Manure exists
				elsif ManureAdded > 1
					begin dialogue
						eject evil spirit
						say HELP_TEXT_SPECIAL_CROPS_340
						wait until read
						say HELP_TEXT_SPECIAL_CROPS_350
						wait until read
						send evil spirit home
					end dialogue
					delete Manure with fade
					wait until not Manure exists
				end if
				ManureAdded++
			end if
		end if
	end while
	if Field == 1
		set CropScoreboard1 score 10
	else
		set CropScoreboard2 score 10
	end if
	StopTheHalfwayMadness = 1
	wait until get distance from (camera position) to [Crop] < 100 and get game time > 8 and get game time < 21
	Fireworks = create special effect SPOT_VISUAL_TOWN_FIREWORKS at [Crop] time 60
	CropCompleted++
	if CropCompleted == 1
		begin dialogue
			eject good spirit
			make good spirit point at [Crop]
			say single line HELP_TEXT_SPECIAL_CROPS_360
			wait until read
			stop good spirit pointing
			send good spirit home
		end dialogue
	else
		begin dialogue
			eject good spirit
			make good spirit point at [Crop]
			say single line HELP_TEXT_SPECIAL_CROPS_370
			wait until read
			stop good spirit pointing
			send good spirit home
		end dialogue
	end if
end script CropStatus
