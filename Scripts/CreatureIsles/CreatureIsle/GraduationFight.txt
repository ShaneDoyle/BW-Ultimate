				CrocShouldAttack = False
				CrocShouldMove = False
				CrocShouldDefend = True
			elsif PlayerRecoiling == True
				CrocShouldAttack = True
				CrocShouldMove = False
				CrocShouldDefend = False
			else
				CrocShouldAttack = True
				CrocShouldMove = False
				CrocShouldDefend = False
			end if
			if CrocShouldAttack == True
				if get ForceAttackTimer time remaining <= 0
					set ForceAttackTimer time to 45 seconds
					if number from 1 to 2 == 1
						DistanceBetweenCreatures = OutOfAttackRange + 1
					end if
				end if
				if DistanceBetweenCreatures > OutOfAttackRange
					if get OutOfRangeAttackTimer time remaining == 0
						if DontUseForcePowerAnyMore == False
							set ForceAttackTimer time to 45 seconds
							DoForcePower = True
							set OutOfRangeAttackTimer time to 15 seconds
						end if
					else
					end if
				elsif DistanceBetweenCreatures > LongAttackRange
					RandomHit = number from 1 to 4
					if RandomHit == 1
						DoHighHit = True
					elsif RandomHit == 2
						DoMidHit = True
					elsif RandomHit == 3
						DoLowHit = True
					else
						if get CanDoSpecial time remaining <= 0
							DoSpecialMove = True
							set CanDoSpecial time to 10 seconds
						else
							DoHighHit = True
						end if
					end if
				elsif DistanceBetweenCreatures > ShortAttackRange
					RandomHit = number from 1 to 3
					if RandomHit == 1
						DoHighHit = True
					elsif RandomHit == 2
						DoMidHit = True
					else
						DoLowHit = True
					end if
				else
					RandomHit = number from 1 to 2
					if RandomHit == 1
						DoHighHit = True
					else
						DoMidHit = True
					end if
				end if
			elsif CrocShouldDefend == True
				DoBlock = True
			elsif CrocShouldMove == True
				if DistanceBetweenCreatures <= ShortAttackRange
					DoRetreat = True
				else
					SideStep = number from 1 to 2
					if SideStep == 1
						DoSideLeft = True
					else
						DoSideRight = True
					end if
				end if
			end if
			if AllHealsDone == False
				if CrocHealth <= 0.75 or CrocHealth <= 0.5 or CrocHealth <= 0.25
					if ThreeQuartersHealDone == False and CrocHealth <= 0.75
						DoHeal = True
						ThreeQuartersHealDone = True
					elsif HalfHealDone == False and CrocHealth <= 0.5
						DoHeal = True
						HalfHealDone = True
					elsif OneQuarterHealDone == False and CrocHealth <= 0.25
						DoHeal = True
						AllHealsDone = True
					end if
					if DoHeal == True
						run script CrocForceHeal
						DoHeal = False
					end if
				end if
			end if
			if DoHighHit == True
				NextFightMoveForCroc = variable C_MOVE_STAND
				Attack = True
			elsif DoMidHit == True
				NextFightMoveForCroc = variable C_MOVE_WALK
				Attack = True
			elsif DoLowHit == True
				NextFightMoveForCroc = variable C_MOVE_RUN
				Attack = True
			elsif DoSpecialMove == True
				NextFightMoveForCroc = variable C_MOVE_R_SPIN_90
				Special = True
			elsif DoBlock == True
				NextFightMoveForCroc = variable C_MOVE_R_SPIN_0
				Defend = True
			elsif DoAdvance == True
				NextStepMoveForCroc = variable C_FIGHT_STEP_FORWARD
				Step = True
			elsif DoRetreat == True
				NextStepMoveForCroc = variable C_FIGHT_STEP_BACK
				Step = True
			elsif DoSideLeft == True
				NextStepMoveForCroc = variable C_FIGHT_STEP_LEFT
				Step = True
			elsif DoSideRight == True
				NextStepMoveForCroc = variable C_FIGHT_STEP_RIGHT
				Step = True
			elsif DoForcePower == True
				RandomForcePower = number from 1 to 4
				if RandomForcePower == 1
					run script CrocForceFlame
				elsif RandomForcePower == 2
					run background script CrocForceInvisible
				elsif RandomForcePower == 3
					run script CrocForceBolt
				elsif RandomForcePower == 4
					run script CrocForceFlame
				end if
			elsif DoSpellCast == True
				NextSpellMoveForCroc = variable SPELL_FIREBALL_LEVEL_1
				Spell = True
			end if
			if Step == True
				queue GCroc fight step constant NextStepMoveForCroc
			elsif Attack == True
				queue GCroc fight move constant NextFightMoveForCroc strength 1
			elsif Defend == True
				queue GCroc fight move constant NextFightMoveForCroc strength 0
			elsif Special == True
				queue GCroc fight move constant NextFightMoveForCroc strength 1
			elsif Spell == True
				queue GCroc fight spell constant NextSpellMoveForCroc
			end if
			if CrocWinning == True
				if MyCreatureHealth < 0.8 and MyCreatureHealth > 0.2
					if get WinLoseTimer time remaining <= 0
						WinLose = variable constant from HELP_TEXT_CROC_FIGHT_CROC_WINNING_01 to HELP_TEXT_CROC_FIGHT_CROC_WINNING_03
						begin dialogue
							queue GCroc fight move C_MOVE_R_SPIN_0 strength 0
							start say sound constant WinLose
							wait until not say sound constant WinLose playing
						end dialogue
						set WinLoseTimer time to 20 seconds
					end if
				end if
			else
				if CrocHealth < 0.8 and CrocHealth > 0.2
					if get WinLoseTimer time remaining <= 0
						WinLose = variable constant from HELP_TEXT_CROC_FIGHT_CROC_LOSING_01 to HELP_TEXT_CROC_FIGHT_CROC_LOSING_03
						begin dialogue
							queue GCroc fight move C_MOVE_R_SPIN_0 strength 0
							start say sound constant WinLose
							wait until not say sound constant WinLose playing
						end dialogue
						set WinLoseTimer time to 20 seconds
					end if
				end if
			end if
		end if
	end while
	clear GCroc fight queue
	clear MyCreature fight queue
	enable GCroc auto fighting
	enable MyCreature auto fighting
end script CrocFightAI

	wait until not CrocRocks[RockNo] exists or not CrocRocks[RockNo] is FLYING or not get distance from [CrocRocks[RockNo]] to [CrocFightArenaPos] < 150
	if CrocRocks[RockNo] exists
		if CrocRocks[RockNo] is FLYING
			CreateRock = 1
		else
			CreateRock = 0
		end if
	else
		CreateRock = 1
	end if
	if CreateRock == 1
		if CrocRocks[RockNo] exists
			delete CrocRocks[RockNo]
		end if
		Scale = number from 1.1 to 2.4
		Angle = number from 0 to 360
		CrocRocks[RockNo] = create with angle Angle and scale Scale ROCK_OBJECT ROCK at [PilePos] + [number from -10 to 10, 0, number from -10 to 10]
	else
		set CrocRocks[RockNo] position to [PilePos] + [number from -10 to 10, 0, number from -10 to 10]
	end if
	enable CrocRocks[RockNo] indestructible
	disable CrocRocks[RockNo] pickup
	SFX = create special effect SPOT_VISUAL_OBJECT_APPEAR at [CrocRocks[RockNo]] time 2
	Smoke = create special effect SPOT_VISUAL_BANG at [CrocRocks[RockNo]] time 2
end script CrocRepositionThrownRock

	end loop
