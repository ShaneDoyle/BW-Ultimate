begin script ScoreRound(RoundNo, WhosBowl)
	PlayerOldBallCount = 0
	PlayerThisScore = 0
	CowOldBallCount = 0
	CowThisScore = 0
	DoneScoring = 0
	UnclaimedPoints = 0
start
	while DoneScoring == 0
		if WhosBowl == 0
			PlayerOldBallCount = PlayerTotalBowled
			wait until PlayerOldBallCount != PlayerTotalBowled
			if PlayerLastScoreBowled == 10
				PlayerOldBallCount = PlayerTotalBowled
				PlayerUnclaimedPoints += 10
				UnclaimedPoints = 10
				wait until PlayerOldBallCount != PlayerTotalBowled
				PlayerThisScore = 10 + PlayerLastScoreBowled
				PlayerUnclaimedPoints += PlayerLastScoreBowled
				UnclaimedPoints += PlayerLastScoreBowled
				PlayerOldBallCount = PlayerTotalBowled
				wait until PlayerOldBallCount != PlayerTotalBowled
				PlayerThisScore += PlayerLastScoreBowled
				PlayerUnclaimedPoints += PlayerLastScoreBowled
				UnclaimedPoints += PlayerLastScoreBowled
			else
				PlayerThisScore = PlayerLastScoreBowled
				PlayerUnclaimedPoints += PlayerLastScoreBowled
				UnclaimedPoints = PlayerLastScoreBowled
				PlayerOldBallCount = PlayerTotalBowled
				wait until PlayerOldBallCount != PlayerTotalBowled
				PlayerUnclaimedPoints += PlayerLastScoreBowled
				UnclaimedPoints += PlayerLastScoreBowled
				if PlayerThisScore + PlayerLastScoreBowled == 10
					PlayerOldBallCount = PlayerTotalBowled
					wait until PlayerOldBallCount != PlayerTotalBowled
					PlayerThisScore = 10 + PlayerLastScoreBowled
					PlayerUnclaimedPoints += PlayerLastScoreBowled
					UnclaimedPoints += PlayerLastScoreBowled
				else
					PlayerThisScore += PlayerLastScoreBowled
				end if
			end if
		else
			CowOldBallCount = CowTotalBowled
			wait until CowOldBallCount != CowTotalBowled
			if CowLastScoreBowled == 10
				CowOldBallCount = CowTotalBowled
				CowUnclaimedPoints += 10
				UnclaimedPoints = 10
				wait until CowOldBallCount != CowTotalBowled
				CowThisScore = 10 + CowLastScoreBowled
				CowUnclaimedPoints += CowLastScoreBowled
				UnclaimedPoints += CowLastScoreBowled
				CowOldBallCount = CowTotalBowled
				wait until CowOldBallCount != CowTotalBowled
				CowThisScore += CowLastScoreBowled
				CowUnclaimedPoints += CowLastScoreBowled
				UnclaimedPoints += CowLastScoreBowled
			else
				CowThisScore = CowLastScoreBowled
				CowUnclaimedPoints += CowLastScoreBowled
				UnclaimedPoints = CowLastScoreBowled
				CowOldBallCount = CowTotalBowled
				wait until CowOldBallCount != CowTotalBowled
				CowUnclaimedPoints += CowLastScoreBowled
				UnclaimedPoints += CowLastScoreBowled
				if CowThisScore + CowLastScoreBowled == 10
					CowOldBallCount = CowTotalBowled
					wait until CowOldBallCount != CowTotalBowled
					CowThisScore = 10 + CowLastScoreBowled
					CowUnclaimedPoints += CowLastScoreBowled
					UnclaimedPoints += CowLastScoreBowled
				else
					CowThisScore += CowLastScoreBowled
				end if
			end if
		end if
		if WhosBowl == 0
			PlayerTotalScore += PlayerThisScore
			PlayerUnclaimedPoints -= UnclaimedPoints
		else
			CowTotalScore += CowThisScore
			CowUnclaimedPoints -= UnclaimedPoints
		end if
		DoneScoring = 1
		until CancelCurrentGame == 1
	end while
	if CancelCurrentGame == 1
		PlayerUnclaimedPoints = 0
		CowUnclaimedPoints = 0
		PlayerTotalScore = 0
		CowTotalScore = 0
		PlayerTotalBowled = 0
		CowTotalBowled = 0
		PlayerLastScoreBowled = 0
		CowLastScoreBowled = 0
		PlayerFirstBallScore = 0
		PlayerSecondBallScore = 0
		CowFirstBallScore = 0
		CowSecondBallScore = 0
		AmountOfPinsHit = 0
	end if
end script ScoreRound

begin script CalculateBestTarget

		AimingPos[2]++
		if PinArray[3] == 1
			AimingPos[2]++
		end if
	end if
	if PinArray[6] == 1
		AimingPos[4]++
		if PinArray[3] == 1
			AimingPos[4]++
		end if
	end if
	if PinArray[2] == 1 or PinArray[8] == 1
		AimingPos[1]++
		if PinArray[5] == 1
			AimingPos[1]++
		end if
	end if
	if PinArray[9] == 1
		AimingPos[3]++
		if PinArray[5] == 1
			AimingPos[3]++
		end if
	end if
	while Counter < 5
		if AimingPos[Counter] > BestValueSoFar
			BestValueSoFar = AimingPos[Counter]
			BestPosSoFar = Counter
		end if
		Counter++
	end while
	TargetArea = BestPosSoFar

	GotNothing = 0
start

				if WhatBowl == FirstBowl
					if PlayerLastScoreBowled == 10
						run script PlayAnims(Player, Strike)
						GotStrike = 1
					elsif PlayerLastScoreBowled == 0
						run script PlayAnims(Player, MissedPins)
						GotNothing = 1
					end if

					if PlayerFirstBallScore + PlayerLastScoreBowled == 10

							GotNothing = 1
						end if

							run script PlayAnims(Player, Spare)
							GotSpare = 1
						elsif PlayerLastScoreBowled + PlayerFirstBallScore == 0
							run script PlayAnims(Player, MissedPins)
							GotNothing = 1
						end if
					end if
				elsif WhatBowl == ExtraBowl
					if PlayerSecondBallScore == 10

						elsif CowLastScoreBowled + CowFirstBallScore == 0
							run script PlayAnims(Cow, MissedPins)
							GotNothing = 1
						end if
					end if

					if CowSecondBallScore == 10
						if CowLastScoreBowled == 10

							GotStrike = 1
						elsif CowLastScoreBowled == 0
							run script PlayAnims(Cow, MissedPins)
							GotNothing = 1
						end if

							run script PlayAnims(Cow, MissedPins)

begin script PlayPinSound
	Counter = 0
	QuitLoops = 0
	SpeedOfBall = 0
	SpeedTimer = create timer for 0 seconds
	NumPins = 0
start
	begin loop
		Counter = 0
		while (Counter < 10 or QuitLoops == 1) and CancelCurrentGame == 0
			if Counter != 4 and Counter != 8 and Counter != 7
				if Pin[Counter] exists and Pin[Counter] is FLYING
					if Counter < 3
						SpeedOfBall = get SpeedTimer time since set
						NumPins = 10 - AmountOfPinsHit
						if SpeedOfBall < 3 and NumPins >= 5
							start sound constant from LH_SAMPLE_BOWLSTRIKE_01 to LH_SAMPLE_BOWLSTRIKE_02 at [Pin[Counter]]
						elsif SpeedOfBall < 4 and NumPins >= 3
							start sound constant from LH_SAMPLE_SKITTLEHITMULTIPLE_01 to LH_SAMPLE_SKITTLEHITMULTIPLE_02 at [Pin[Counter]]
						elsif SpeedOfBall < 5
							start sound constant from LH_SAMPLE_SKITTLEHITSINGLE_01 to LH_SAMPLE_SKITTLEHITSINGLE_03 at [Pin[Counter]]
						end if
					else
						start sound constant from LH_SAMPLE_SKITTLEHITSINGLE_01 to LH_SAMPLE_SKITTLEHITSINGLE_03 at [Pin[Counter]]
					end if
					QuitLoops = 1
				end if
			end if
			Counter++
			until CancelCurrentGame == 1
		end while
		until not BowlingBall is FLYING or not BowlingBall exists or not get distance from [BowlingBall] to [BBallPos] < 90
		until QuitLoops == 1
		until CancelCurrentGame == 1
	end loop
end script PlayPinSound
